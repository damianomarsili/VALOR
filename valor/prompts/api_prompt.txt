    ----------

def gd_detect(img_pth, prompt):
    """
    Run object detection on an image and return the post-processed bounding boxes.
    Not necessarily in the same order as the prompt.

    Parameters
    ----------
    img_pth : str
        Path to the input image file.
    prompt  : str
        Natural-language description of the object to be detected that describes the object's appearance. It can be a noun, e.g., "fireplace, coffee table, sofa", or accompanied by an appearance attribute, e.g., "the black coffee table", "the pink pillow". 
        NOTE: do not pass spatial relationship attribues, e.g., "the rightmost sofa" is wrong -- they don't relate to the object's appearance.

    Returns
    -------
    list[dict]:
        A list where each element is a dict {"bbox": [x1, y1, x2, y2], "label": <str>}. The list is NOT necessarily in order of the prompt.
    """
